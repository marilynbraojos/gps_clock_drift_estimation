# https://pytorch-forecasting.readthedocs.io/en/stable/api/pytorch_forecasting.data.timeseries.TimeSeriesDataSet.html
training = TimeSeriesDataSet(
    data[lambda x: x.time_idx <= training_cutoff],
    time_idx="time_idx",
    target="Clock Bias (seconds)",
    group_ids=["SV Name"],
    max_encoder_length=max_encoder_length,
    max_prediction_length=max_prediction_length,
    static_categoricals=[],
    static_reals=["SV Health", "SV Name"],
    time_varying_known_categoricals=[],
    time_varying_known_reals=[],
    time_varying_unknown_categoricals=[],
    time_varying_unknown_reals=["Clock Bias (seconds)",  "Clock Drift", "Clock Drift Rate", "Crs", "Delta n", "M0", "Cuc",
        "Eccentricity", "Cus", "sqrtA", "Cic", "Omega0", "Cis", "Io",
        "Crc", "omega", "OmegaDot", "SV Accuracy"],
    add_relative_time_idx=True,
    add_target_scales=True,
    allow_missing_timesteps=False,
)

# create validation set (predict=True) which means to predict the last max_prediction_length points in time
# for each series

    training, data, predict=True, stop_randomization=True
)

val_dataloader = validation.to_dataloader(
